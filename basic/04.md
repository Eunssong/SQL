# GROUP BY
- GROUP BY절을 사용하여 그룹 수준 추가
- 나열한 컬럼에 따라 그룹화 생성
  
  SELECT    
      \<aggregate function>\(\<col1>)...\<colN>    
  FROM \[table]        
  GROUP BY \<col1>, \<col2>       
  ORDER BY \<col1>, \<col2>

  SELECT    
      industry,   
      count(*) as cnt    
  FROM nasdaq_company    
  GROUP BY industry
  
# HAVING
- GROUP BY 결과에 대한 필터 기능
- SELECT의 WHERE와 유사한 기능

SELECT    
      \<aggregate function>\(\<col1>)...\<colN>    
  FROM \[table]      
  WHERE \<col> = \[value]
  GROUP BY \<col1>, \<col2>       
  HAVING \<aggregate function>\(col1) = \<value>     

    SELECT    
      industry,   
      count(*) as cnt    
  FROM nasdaq_company    
  GROUP BY industry     
  HAVING count(*) >= 100
